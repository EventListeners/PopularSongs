import React from 'react';
import { Popover, PopoverBody } from 'reactstrap';
   
 
class Song extends React.Component {
 
    constructor(props) {
        super(props);
 
        this.toggle = this.toggle.bind(this);
        
        this.state = {
            playing: false,
            moreOptions: false,
            hover: false,
            popoverOpen: false
        };
    }   
      
    displayNumber () {
        return (<div id={"number"}>{this.props.counter}</div>);
    }
 
    displayPlay () {
        return (<ion-icon id={"play"} name={"ios-play"}></ion-icon>);
    }
 
    toggle() {
        this.setState({popoverOpen: !this.state.popoverOpen});
    }
        
    render () {
 
        //library={e[1].library} playing={e[1].library}
         
        return (
           <div className={"row song"} onMouseOver={() => {this.setState({hover: true})}} onMouseOut={() => this.setState({hover: false})}>
                <div className={"col col-lg-1"}></div>
 
                <div className={"col col-lg-1"}>
                    <img src={this.props.albumURL} alt={"artist album cover"} width={"50"}/>
                </div>
 
                <div className={"song-number-play col col-lg-1"}>
                
                {this.state.hover ? this.displayPlay() : this.displayNumber()}
                    
                </div>
 
                <div className={"col col-lg-1"}>
                    <ion-icon className={"plus"} name={"ios-add"}></ion-icon>
                </div>
 
                <div className={"col col-lg-4"}>{this.props.songName}</div>
 
                <div className={"col col-lg-1"}>
                    <ion-icon id="Popover1" className="more" name={"ios-more"} onClick={this.toggle}></ion-icon>
                </div>
 
                <div className={"col col-lg-2"}>{this.props.streams.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",")}</div>
 
                <div className={"col col-lg-1"}></div>
 
 
                <div id={"more-options"}>
                    <Popover placement="bottom" isOpen={this.state.popoverOpen} target="Popover1" toggle={this.toggle}>
                        <PopoverBody>
                            <table>
                                <tbody>
                                    <tr><td>Add to Queue</td></tr>
                                    <tr><td>Go to Song Radio</td></tr>
                                    <tr><td>Go to Artist</td></tr>
                                    <tr><td>Go to Album</td></tr>
                                    <tr><td>Show Credits</td></tr>
                                    <tr><td>Save to Your Library</td></tr>
                                    <tr><td>Add to Playlist</td></tr>
                                    <tr><td>Share</td></tr>
                                </tbody>
                            </table>
                        </PopoverBody>
                    </Popover>
                </div>
            </div>
            
        );
    }
 
}
      
export default Song;
